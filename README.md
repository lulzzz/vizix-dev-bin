1. Add bin folder to SO's PATH

2. create vizix network
      > create-network

3. create folders:
      > mkdir ~/docker-volumes/mysql-data
      > mkdir ~/docker-volumes/mysql-data

4. create basic services
      > run-mysql
      > run-mongo
      > run-mosquitto
      > run-hazelcast
5. create services
      > cd <path_to_services>
      > build-services
      > run-services

6. create ui
      > cd <path_to_ui>
      > build-ui
      > run-ui
